Write a code to calculate area of a rectangle using inner function. 
    function area(l, b){
        this.l = l;
        this.area = function (){
             this.b = b;
             return (this.l * this.b) ;
        };
    };
    var c = new area(6,5);
    console.log('Area =', c.area());
