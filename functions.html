//Write a function that accepts parameter n and returns factorial of n. 
    function factorial(n){
        if(n===0){
            return "1";//doesn't return 0.returns 1.
        }
        if(n===1){
            return n;
        }
        return n*factorial(n-1);//4*(4-1)*(3-1)*(2-1)*(1-1)//0 returns 1
    }
    console.log(factorial(4));
